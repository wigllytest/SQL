Weather Observation Station 3
SELECT DISTINCT CITY 
FROM STATION 
WHERE MOD(STATION.ID,2)=0;

Weather Observation Station 4
SELECT COUNT(CITY) - COUNT(DISTINCT CITY) AS N
FROM STATION;

Weather Observation Station 5
SELECT * FROM
((SELECT * FROM 
(select CITY, LENGTH(CITY) K FROM STATION 
WHERE LENGTH(CITY) = 
(SELECT MIN(LENGTH(CITY)) FROM STATION ) ORDER BY CITY ASC ) WHERE ROWNUM=1 ) UNION 
(SELECT * FROM ( select CITY, LENGTH(CITY) K FROM STATION WHERE LENGTH(CITY) =
(SELECT MAX(LENGTH(CITY)) FROM STATION ) ORDER BY CITY ASC ) WHERE ROWNUM=1 ) );

Weather Observation Station 6
SELECT DISTINCT CITY 
FROM STATION 
WHERE REGEXP_LIKE(LOWER(CITY), '^[aeiou]') ;
